\frame {
	\frametitle{\underline{Introduction to Cayley Graphs}}
	\only<1>{
	\begin{itemize}
                  \item<1-> For any positive integer $d$,
\[
m(d,A) =\ max\{m | diam(Cay(Z_m,A)) \leq d\},
\]

		\item<1->For positive integers $d$ and $k$,
\[
m(d,k) = max\{m(d,A) | \text{ there exists a set A with } |A| = k \},
\]


	\end{itemize}
}
}
